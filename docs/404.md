<script>
(function() {
  const path = window.location.pathname;
  const base = '/Snap.Hutao-Manjusaka.Docs/';
  
  // Remove base from path
  let cleanPath = path;
  if (cleanPath.startsWith(base)) {
    cleanPath = cleanPath.substring(base.length);
  }
  
  // If path already starts with a language code, do nothing (show 404)
  const langCodes = ['zh', 'en', 'ru', 'id', 'jp', 'tw'];
  const pathParts = cleanPath.split('/');
  if (langCodes.includes(pathParts[0])) {
    return; // Already in a language path, show 404
  }
  
  // Detect browser language
  const lang = navigator.language.split('-')[0];
  const supportedLangs = ['zh', 'en', 'ru', 'id', 'jp', 'tw'];
  const targetLang = supportedLangs.includes(lang) ? lang : 'zh';
  
  // Redirect to the language-specific path
  const newPath = base + targetLang + '/' + cleanPath;
  window.location.href = newPath;
})();
</script>
